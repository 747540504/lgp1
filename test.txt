public class test{
	public static void main(String[] args){
		int sum=0;
		for(int year=1900;year<2016;year++){
			if(year%4==0&&year%100!=0||year%400==0){
				sum=sum+366;
			}else{
				sum=sum+365;
			}
		}
		for(int month=1;month<=3;month++){
			if(month==2){
				if(2016%4==0&&2016%100!=0||2016%400==0){
					sum=sum+29;
				}else{
					sum=sum+28;
				}
			}if(month==4||month==6||month==9||month==11){
				sum=sum+30;
				}else{
				sum+=31;
				}
			}
		}
		sum+=29;
		System.out.println("1900年1月1日到2016年3月29日一共有"+sum+"天");
	}
}
