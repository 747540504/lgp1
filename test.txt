package java1;

import java.util.*;

public class java01 {
	public static void main(String[] args){
		Scanner in=new Scanner(System.in);
		System.out.println("请输入数字：");
		int number=in.nextInt();
		boolean tag=false;
		int i,j;
		for(i=2;i<=number/2;i++){
			if(number%i==0){
				tag=true;
				break;
			}
		}
		if(tag==true)
			System.out.println(number+"不是素数");
		else
			System.out.println(number+"是素数");
	}
}
